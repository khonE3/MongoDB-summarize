📊 Aggregation Framework

          
Pipeline สำหรับประมวลผลและวิเคราะห์ข้อมูล


          

            
Pipeline Flow

            

              

                🔍
                $match
                กรองข้อมูล
              

              
→

              

                📦
                $group
                จัดกลุ่ม
              

              
→

              

                ↕️
                $sort
                เรียงลำดับ
              

              
→

              

                📋
                $project
                เลือก field
              

              
→

              

                ✅
                Result
                ผลลัพธ์
              

            

          


          

            

              
รายงานยอดขายตามหมวด

              
            

            

              
db.orders.aggregate([
  // Stage 1: กรองเฉพาะออเดอร์สำเร็จ
  { $match: { status: "completed" } },

  // Stage 2: จัดกลุ่มตาม category
  { $group: {
    _id: "$category",
    totalRevenue: { $sum: "$amount" },
    orderCount: { $sum: 1 },
    avgOrder: { $avg: "$amount" }
  }},

  // Stage 3: เรียงตามยอดมากสุด
  { $sort: { totalRevenue: -1 } },

  // Stage 4: จัดรูปแบบผลลัพธ์
  { $project: {
    _id: 0,
    category: "$_id",
    totalRevenue: { $round: ["$totalRevenue", 2] },
    orderCount: 1,
    avgOrder: { $round: ["$avgOrder", 2] }
  }}
])

            

            

              
📋 Result

              

            

          


          

            

              
$lookup — JOIN กับ collection อื่น

              
            

            

              
db.orders.aggregate([
  // JOIN orders กับ users
  { $lookup: {
    from: "users",
    localField: "userId",
    foreignField: "_id",
    as: "customer"
  }},

  // แตก array customer
  { $unwind: "$customer" },

  // เลือก field ที่ต้องการ
  { $project: {
    orderId: "$_id",
    product: 1,
    total: 1,
    customerName: "$customer.name",
    customerEmail: "$customer.email"
  }}
])

            

            

              
📋 Result

              

            

          


          

            

              
$facet — Pipeline หลายสายพร้อมกัน

              
            

            

              
db.products.aggregate([
  { $facet: {
    // สาย 1: สถิติราคา
    priceStats: [
      { $group: {
        _id: null,
        avgPrice: { $avg: "$price" },
        maxPrice: { $max: "$price" },
        minPrice: { $min: "$price" }
      }}
    ],
    // สาย 2: นับตามหมวด
    categoryBreakdown: [
      { $group: {
        _id: "$category",
        count: { $sum: 1 }
      }},
      { $sort: { count: -1 } }
    ],
    // สาย 3: Top 3 สินค้าแพงสุด
    topExpensive: [
      { $sort: { price: -1 } },
      { $limit: 3 },
      { $project: { name: 1, price: 1 } }
    ]
  }}
])

            

            

              
📋 Result

Single Document Atomicity

              
            

            

              
// ทั้ง $inc และ $push จะสำเร็จหรือล้มเหลวพร้อมกัน
db.accounts.updateOne(
  { _id: "account1" },
  {
    $inc: { balance: -100 },

Multi-Document Transaction (โอนเงิน)

              
            

            

              
const session = db.getMongo().startSession()
session.startTransaction()

try {
  const accounts = session.getDatabase("bank").accounts

  // หักเงินจากบัญชี A
  accounts.updateOne(
    { _id: "accountA" },
    { $inc: { balance: -500 } },
    { session }
  )

  // เพิ่มเงินให้บัญชี B
  accounts.updateOne(
    { _id: "accountB" },
    { $inc: { balance: 500 } },
    { session }
  )

  // ✅ ทั้ง 2 operations สำเร็จ → commit
  session.commitTransaction()
} catch (error) {
  // ❌ เกิดข้อผิดพลาด → rollback ทั้งหมด
  session.abortTransaction()
} finally {
  session.endSession()
}

import mongoose from 'mongoose'

// เชื่อมต่อ
await mongoose.connect('mongodb://localhost:27017/myApp')

// สร้าง Schema
const userSchema = new mongoose.Schema({
  name:  { type: String, required: true, trim: true },
  email: { type: String, required: true, unique: true,
           lowercase: true },
  age:   { type: Number, min: 0, max: 150 },
  role:  { type: String, enum: ['user','admin'],
           default: 'user' },
  tags:  [String],
  profile: {
    avatar: String,
    bio:    { type: String, maxlength: 500 }
  }
}, {
  timestamps: true  // adds createdAt, updatedAt
})

// สร้าง Model
const User = mongoose.model('User', userSchema)

Middleware & Virtuals

              
            

            

              
// Pre-save middleware
userSchema.pre('save', async function(next) {
  if (this.isModified('password')) {
    this.password = await bcrypt.hash(this.password, 12)
  }
  next()
})

// Virtual field (ไม่ได้เก็บใน DB)
userSchema.virtual('fullName').get(function() {
  return `${this.firstName} ${this.lastName}`
})

// Static method
userSchema.statics.findByEmail = function(email) {
  return this.findOne({ email: email.toLowerCase() })
}

// Instance method
userSchema.methods.isAdmin = function() {
  return this.role === 'admin'

createUser() — สร้าง User

                
              

              

                
// สร้าง Admin User
use admin
db.createUser({
  user: "admin",
  pwd: "strongPassword123!",
  roles: [
    { role: "userAdminAnyDatabase", db: "admin" },
    { role: "readWriteAnyDatabase", db: "admin" }
  ]
})

// สร้าง User สำหรับ App
use myApp
db.createUser({
  user: "appUser",
  pwd: "appPassword",
  roles: [
    { role: "readWrite", db: "myApp" },
    { role: "read", db: "reporting" }
  ]
})

User Management — จัดการ Users

                
              

              

                
// ดู Users ทั้งหมดใน database
db.getUsers()

// ดูข้อมูล User เฉพาะ
db.getUser("appUser")

// เปลี่ยนรหัสผ่าน
db.changeUserPassword("appUser", "newSecurePassword!")

// อัปเดต User (เช่น เปลี่ยน role)
db.updateUser("appUser", {
  roles: [
    { role: "readWrite", db: "myApp" },
    { role: "readWrite", db: "analytics" }
  ]
})

// เพิ่ม Role ให้ User
db.grantRolesToUser("appUser", [
  { role: "dbAdmin", db: "myApp" }
])

// ถอน Role จาก User
db.revokeRolesFromUser("appUser", [
  { role: "dbAdmin", db: "myApp" }
])

// ลบ User
db.dropUser("appUser")

              

              

                
📋 Result

                

              

            

          


          

            

              
🛡️ Built-in Roles

              

                
                  
Roleสิทธิ์ใช้เมื่อ

                
                
                  
readอ่านข้อมูลReport / Analytics

                  
readWriteอ่าน + เขียนApplication User

                  
dbAdminจัดการ DB (index, stats)DBA

                  
userAdminจัดการ UsersAdmin

                  
clusterAdminจัดการ ClusterDevOps

                  
backupBackup ข้อมูลBackup System

                  
rootสิทธิ์ทั้งหมดSuper Admin

                
              

            


            

              

                
Connection String พร้อม Authentication

                
              

              

                
// Local connection with auth
mongosh "mongodb://appUser:password@localhost:27017/myApp"

// Atlas connection
mongosh "mongodb+srv://appUser:password@cluster0.xxxxx.mongodb.net/myApp"

// Connection String format
mongodb://[username:password@]host[:port]/[database][?options]

// ตัวอย่างกับ options
mongodb://admin:pass@localhost:27017/myApp?authSource=admin&retryWrites=true
